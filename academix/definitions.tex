% ----------------------------------------------------------------------------
% definitions.tex
%
% This file contains the main formatting definitions for the document class.
% It includes the following key formatting elements:
%   - spacing settings (line spacing, paragraph indentation, etc.)
%   - margin sizes (top, bottom, left, right, and other custom margins)
%   - minipage settings (positioning, width, and spacing)
%   - cover page and back cover margin adjustments
%   - caption and appendix formatting (spacing, margins, symbols)
%   - chapter title and section title spacing adjustments
%   - field titles (Acknowledgements, Abstract, Resumo, Advisor, etc.)
%   - commands for customization of degree information and text blocks
% ----------------------------------------------------------------------------

% ----------------------------------------------------------------------------
% FORMATTING DEFINITIONS
% ----------------------------------------------------------------------------

% Line spacing definitions (single, one-and-a-half, and double)
\newcommand*{\singleSpacingRate}{1.0}
\newcommand*{\oneHalfSpacingRate}{1.5}
\newcommand*{\doubleSpacingRate}{2.0}

% Paragraph indentation and skip (space before new paragraphs)
\newcommand*{\indentationSize}{1.25cm}
\newcommand*{\paragraphSkip}{0cm}

% Separator symbol used in figure/table captions and appendices
\newcommand{\captionSeparator}{ - }
\newcommand{\anApSeparator}{-}

% Margin sizes
\newcommand{\topMargin}{3.0cm}
\newcommand{\bottomMargin}{2.0cm}
\newcommand{\leftMargin}{3.0cm}
\newcommand{\rightMargin}{2.0cm}

% Distance from the top of the header to the top of the text
\newcommand{\headerToText}{1.0cm}

% Spacing adjustments between title page elements
\newcommand{\authorToTitle}{9cm}  % Space between author and title
\newcommand{\authorMargin}{1.3cm} % Offset from page to author in cover

% Minipage positioning and size
\newcommand{\minipagePosition}{0.5}       % Horizontal position of minipage
\newcommand{\minipageWidth}{0.45}         % Width of the minipage
\newcommand{\minipageTopMargin}{1.7cm}    % Top margin of the minipage

% Paragraph spacing adjustments for minipage
\newcommand{\minipageParagraphSpacingA}{0.8cm} % Space between paragraphs (type A)
\newcommand{\minipageParagraphSpacingB}{0.2cm} % Space between paragraphs (type B)

% Chapter title spacing adjustments
\newcommand{\chapterTitleTopMargin}{1cm}
\newcommand{\chapterTitleBottomMargin}{1cm}
\newcommand{\offsetAfterChapter}{-1cm} % Necessary offset after chapter title

% Annex/Appendix title spacing adjustments (in points)
\newcommand{\anApTitleTopMargin}{0pt}
\newcommand{\anApTitleBottomMargin}{1cm}

% Command to insert a minipage block with text, useful for title pages
\newcommand{\minipageText}[1]{\comment{\vspace{1.0cm} #1}}

% Field titles for various sections (can be customized or translated)
\newcommand{\acknowledgmentName}{Acknowledgment}
\newcommand{\abstractName}{Abstract}
\newcommand{\secondAbstractName}{Résumé}
\newcommand{\advisorTitle}{Advisor}
\newcommand{\coadvisorTitle}{Co-Advisor}
\newcommand{\concentrationField}{Concentration Field}
\newcommand{\listOfAbbreviationName}{List of Abbreviations}
\newcommand{\listOfSymbolsName}{List of Symbols}
\newcommand{\listOfFiguresName}{List of Figures}
\newcommand{\listOfTablesName}{List of Tables}
\newcommand{\referenceName}{References}
\newcommand{\appendixName}{Appendix}
\newcommand{\annexName}{Annex}
